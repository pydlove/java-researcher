长期解决方案

1. 磁盘监控告警

# 监控脚本示例 #!/bin/bash THRESHOLD=85 CURRENT_USAGE=$(df / | awk 'NR==2 {print $5}' | sed 's/%//') if [ $CURRENT_USAGE -gt $THRESHOLD ]; then echo "警告: 磁盘使用率 ${CURRENT_USAGE}%" # 发送告警邮件/短信 fi

2. 定期维护脚本

#!/bin/bash # 每日清理脚本，清理30天前的日志 find /app/logs -name "*.log.*" -mtime +30 -delete # 清理/tmp目录 find /tmp -type f -mtime +7 -delete # 清理Tomcat临时文件 find /opt/tomcat/work -name "*" -mtime +1 -delete echo "$(date): 磁盘清理完成" >> /var/log/cleanup.log
